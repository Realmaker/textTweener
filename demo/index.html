This Demo's Code

import { TextTweener } from 'texttweener';

const tt = new TextTweener('#tweener', {
  duration: 4000,
  transitionDuration: 800,
  stagger: 15,
  easing: 'cubic-bezier(0.16, 1, 0.3, 1)',
  textAlign: 'left',
  texts: [
    'Über 30 Sonderzeichen: Klammern (rund), [eckig] & {geschweifte} — alles möglich!',
    'Preise: ab 9,99 € pro Monat. Fragen? E-Mail an info@example.de schreiben!',
    'Mathematik: 2 + 2 = 4, aber 100% der Lösung liegt im Detail... #nerdig',
    'Sprüche & Zitate: "Wer nicht wagt, der nicht gewinnt" — Omas Weisheit für\'s Leben.',
  ],
});

// API
tt.play();                    // Start playback
tt.pause();                   // Pause
tt.next();                    // Next text (returns Promise)
tt.prev();                    // Previous text
tt.goTo(2);                   // Jump to index
tt.destroy();                 // Clean up

// Events
tt.on('beforeTransition', (from, to) => {
  console.log(`Transitioning ${from} → ${to}`);
});

tt.on('change', (index) => {
  console.log(`Now showing text ${index}`);
});
